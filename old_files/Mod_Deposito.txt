Option Explicit

Private s_ID_Istanza_Selezionato As Long
Private s_Provenienza_Selezione As String
Private s_SlotID_Selezionato As Long

Public Sub ApriInterfacciaDeposito(BoxNumero As Integer)
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, sld As Slide
    Dim squadraSheet As Object, depositoSheet As Object, statoSheet As Object
    Dim activePlayerID As Long
    
    Select Case BoxNumero: Case 1: Set sld = ActivePresentation.Slides(34): Case 2: Set sld = ActivePresentation.Slides(35): Case 3: Set sld = ActivePresentation.Slides(36): Case Else: Exit Sub: End Select
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application"): Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True)
    
    Set statoSheet = xlWorkbook.Sheets("Stato_Giocatore")
    activePlayerID = statoSheet.Columns("D").Find(what:=1, lookIn:=-4163, lookAt:=1).Row - 1
    
    If activePlayerID = 1 Then
        Set squadraSheet = xlWorkbook.Sheets("Giocatore1_Squadra")
        Set depositoSheet = xlWorkbook.Sheets("Giocatore1_Deposito")
    Else
        Set squadraSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
        Set depositoSheet = xlWorkbook.Sheets("Giocatore2_Deposito")
    End If
    
    Call PulisciUI_Deposito(sld)
    
    ' Popola la squadra
    Dim i As Long
    For i = 1 To 6
        If i + 1 <= squadraSheet.Cells(squadraSheet.Rows.count, "A").End(-4162).Row Then
            Call PopolaSlot(sld, i, "Attivo", squadraSheet.Cells(i + 1, "B").value, squadraSheet.Cells(i + 1, "C").value, squadraSheet.Cells(i + 1, "D").value)
        End If
    Next i
    
    ' Popola il box del deposito
    Dim startRow As Long, endRow As Long, rigaDeposito As Long
    startRow = (BoxNumero - 1) * 35 + 2
    endRow = startRow + 34
    
    For rigaDeposito = startRow To endRow
        If rigaDeposito <= depositoSheet.Cells(depositoSheet.Rows.count, "A").End(-4162).Row Then
            Dim slotIndex As Long: slotIndex = (rigaDeposito - 2) Mod 35 + 1
            Call PopolaSlot(sld, slotIndex, "Deposito", depositoSheet.Cells(rigaDeposito, "B").value, "", 0)
        End If
    Next rigaDeposito
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
    s_ID_Istanza_Selezionato = 0
End Sub

Private Sub PopolaSlot(sld As Slide, slotIndex As Long, tipo As String, idBase As Long, nome As String, livello As Integer)
    Dim spriteShape As Shape, testoShape As Shape, slotShape As Shape
    On Error Resume Next
    
    Set spriteShape = sld.Shapes("Sprite_Small").GroupItems("Sprite_Small_" & idBase)
    
    If tipo = "Attivo" Then
        Set slotShape = sld.Shapes("Pokemon_Attivo_" & slotIndex).GroupItems("Slot_Attivo_" & slotIndex)
        Set testoShape = sld.Shapes("Pokemon_Attivo_" & slotIndex).GroupItems("Testo_Attivo_" & slotIndex)
        testoShape.TextFrame.TextRange.text = UCase(nome) & " LV. " & livello
    Else ' Deposito
        Set slotShape = sld.Shapes("Slot_Deposito").GroupItems("Slot_" & slotIndex)
    End If
    
    If Not spriteShape Is Nothing And Not slotShape Is Nothing Then
        spriteShape.Visible = msoTrue
        spriteShape.Left = slotShape.Left + (slotShape.Width / 2) - (spriteShape.Width / 2)
        spriteShape.Top = slotShape.Top + (slotShape.Height / 2) - (spriteShape.Height / 2)
    End If
    On Error GoTo 0
End Sub

Public Sub PulisciUI_Deposito(sld As Slide)
    On Error Resume Next
    Dim i As Integer, shp As Shape
    For Each shp In sld.Shapes("Sprite_Small").GroupItems: shp.Visible = msoFalse: Next shp
    For i = 1 To 6
        sld.Shapes("Pokemon_Attivo_" & i).GroupItems("Testo_Attivo_" & i).TextFrame.TextRange.text = ""
        sld.Shapes("Pokemon_Attivo_" & i).GroupItems("Slot_Attivo_" & i).Line.ForeColor.RGB = RGB(255, 255, 255)
    Next i
    For i = 1 To 35
         sld.Shapes("Slot_Deposito").GroupItems("Slot_" & i).Line.ForeColor.RGB = RGB(255, 255, 255)
    Next i
    On Error GoTo 0
End Sub

Public Sub GestisciClickSlot(SlotID As Long, Provenienza As String)
    Dim sld As Slide: Set sld = ActivePresentation.SlideShowWindow.View.Slide: If sld Is Nothing Then Exit Sub
    Dim boxCorrente As Integer: boxCorrente = sld.slideIndex - 33
    
    Dim idIstanzaCliccato As Long
    idIstanzaCliccato = GetIstanzaDaSlot(SlotID, Provenienza, boxCorrente)
    If idIstanzaCliccato = 0 Then
        ' Clic su slot vuoto, resetta la selezione
        EvidenziaSlot sld, s_SlotID_Selezionato, s_Provenienza_Selezione, False
        s_ID_Istanza_Selezionato = 0
        Exit Sub
    End If
    
    If s_ID_Istanza_Selezionato = 0 Then
        ' PRIMO CLIC
        s_ID_Istanza_Selezionato = idIstanzaCliccato
        s_Provenienza_Selezione = Provenienza
        s_SlotID_Selezionato = SlotID
        EvidenziaSlot sld, SlotID, Provenienza, True
    Else
        ' SECONDO CLIC
        EvidenziaSlot sld, s_SlotID_Selezionato, s_Provenienza_Selezione, False
        
        If s_ID_Istanza_Selezionato <> idIstanzaCliccato Then
            ' Esegui lo scambio, che ora gestisce tutti i casi
            If EseguiScambioDati(s_ID_Istanza_Selezionato, s_Provenienza_Selezione, idIstanzaCliccato, Provenienza) Then
                Call ApriInterfacciaDeposito(boxCorrente) ' Ricarica UI dopo scambio
            Else
                ' Questo messaggio ora apparirà solo se lo scambio fallisce per un errore grave
                MsgBox "Errore durante lo scambio!", vbCritical
            End If
        End If
        
        ' Resetta la selezione in ogni caso
        s_ID_Istanza_Selezionato = 0
    End If
End Sub

Private Function GetIstanzaDaSlot(SlotID As Long, Provenienza As String, BoxNumero As Integer) As Long
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim targetSheet As Object, activePlayerID As Long, riga As Long
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application"): Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True)
    
    ' CORREZIONE: Determina il giocatore attivo qui
    activePlayerID = xlWorkbook.Sheets("Stato_Giocatore").Columns("D").Find(what:=1, lookIn:=-4163, lookAt:=1).Row - 1
    
    If Provenienza = "SQUADRA" Then
        If activePlayerID = 1 Then Set targetSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set targetSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
        riga = SlotID + 1
    Else ' DEPOSITO
        If activePlayerID = 1 Then Set targetSheet = xlWorkbook.Sheets("Giocatore1_Deposito") Else Set targetSheet = xlWorkbook.Sheets("Giocatore2_Deposito")
        riga = (BoxNumero - 1) * 35 + SlotID + 1
    End If
    
    On Error Resume Next
    GetIstanzaDaSlot = CLng(targetSheet.Cells(riga, "A").value)
    On Error GoTo 0
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
End Function

Private Sub EvidenziaSlot(sld As Slide, SlotID As Long, Provenienza As String, stato As Boolean)
    Dim slotShape As Shape, colore As Long
    If stato Then colore = RGB(255, 255, 0) Else colore = RGB(255, 255, 255)
    On Error Resume Next
    If Provenienza = "SQUADRA" Then
        Set slotShape = sld.Shapes("Pokemon_Attivo_" & SlotID).GroupItems("Slot_Attivo_" & SlotID)
    Else ' DEPOSITO
        Set slotShape = sld.Shapes("Slot_Deposito").GroupItems("Slot_" & SlotID)
    End If
    If Not slotShape Is Nothing Then slotShape.Line.ForeColor.RGB = colore
    On Error GoTo 0
End Sub

Private Function EseguiScambioDati(id1 As Long, prov1 As String, id2 As Long, prov2 As String) As Boolean
    ' Logica di scambio finale e robusta
    If prov1 = "DEPOSITO" And prov2 = "DEPOSITO" Then
        MsgBox "Per riordinare il deposito, sposta prima un Pokémon in squadra.", vbInformation
        EseguiScambioDati = False
        Exit Function
    End If
    
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim p1_sheet As Object, p2_sheet As Object, p1_riga As Range, p2_riga As Range
    Dim p1_dati As Variant, p2_dati As Variant, activePlayerID As Long
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile)
    
    activePlayerID = xlWorkbook.Sheets("Stato_Giocatore").Columns("D").Find(what:=1, lookIn:=-4163, lookAt:=1).Row - 1
    
    ' Determina i fogli per il Pokémon 1
    If prov1 = "SQUADRA" Then
        If activePlayerID = 1 Then Set p1_sheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set p1_sheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    Else ' prov1 = "DEPOSITO"
        If activePlayerID = 1 Then Set p1_sheet = xlWorkbook.Sheets("Giocatore1_Deposito") Else Set p1_sheet = xlWorkbook.Sheets("Giocatore2_Deposito")
    End If
    
    ' Determina i fogli per il Pokémon 2
    If prov2 = "SQUADRA" Then
        If activePlayerID = 1 Then Set p2_sheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set p2_sheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    Else ' prov2 = "DEPOSITO"
        If activePlayerID = 1 Then Set p2_sheet = xlWorkbook.Sheets("Giocatore1_Deposito") Else Set p2_sheet = xlWorkbook.Sheets("Giocatore2_Deposito")
    End If
    
    ' Trova le righe dei Pokémon nei rispettivi fogli
    Set p1_riga = p1_sheet.Columns("A").Find(what:=id1, lookIn:=-4163, lookAt:=1)
    Set p2_riga = p2_sheet.Columns("A").Find(what:=id2, lookIn:=-4163, lookAt:=1)
    
    If p1_riga Is Nothing Or p2_riga Is Nothing Then
        CloseDatabaseWorkbook xlApp, xlWorkbook, False: EseguiScambioDati = False: Exit Function
    End If
    
    ' Scambia i dati
    p1_dati = p1_riga.Resize(1, 6).value
    p2_dati = p2_riga.Resize(1, 6).value
    
    p1_riga.Resize(1, 6).value = p2_dati
    p2_riga.Resize(1, 6).value = p1_dati
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, True
    EseguiScambioDati = True
End Function
