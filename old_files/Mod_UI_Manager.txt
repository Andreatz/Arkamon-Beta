Public Sub AbilitaPulsantiPerTurno(sld As Slide, turno As String, tipoBattaglia As String, idBaseA As Long, idBaseB As Long)
    On Error Resume Next
    Dim mosseA_Visibili As Boolean, mosseB_Visibili As Boolean
    Dim catturaVisibile As Boolean, avversarioVisibile As Boolean
    
    If UCase$(tipoBattaglia) = "PVP" Then
        mosseA_Visibili = (turno = "A")
        mosseB_Visibili = (turno = "B")
        catturaVisibile = False
        avversarioVisibile = False
    Else
        mosseA_Visibili = (turno = "A")
        mosseB_Visibili = False
        catturaVisibile = (turno = "A" And tipoBattaglia = "Selvatico")
        avversarioVisibile = (turno = "B")
    End If
    
    ' Applica la visibilità ai gruppi principali
    sld.Shapes("Mosse_A").Visible = mosseA_Visibili
    sld.Shapes("Mosse_B").Visible = mosseB_Visibili
    sld.Shapes("Pulsante_Cattura").Visible = catturaVisibile
    sld.Shapes("Pulsante_Turno_Avversario").Visible = avversarioVisibile
    
    ' AGGIORNAMENTO: Chiama AggiornaPulsantiMossa solo se un set di mosse è visibile
    If mosseA_Visibili Then
        Call AggiornaPulsantiMossa(sld, idBaseA, "A")
    End If
    If mosseB_Visibili Then
        Call AggiornaPulsantiMossa(sld, idBaseB, "B")
    End If

    ' Se nessun set di mosse è visibile, nascondi tutte le icone per sicurezza
    If Not mosseA_Visibili And Not mosseB_Visibili Then
        Call GestisciVisibilitaIconeMosse(sld, False)
    End If
    
    On Error GoTo 0
End Sub

Public Sub AggiornaInterfacciaBattaglia_DaTrigger()
    On Error GoTo GestioneErrori
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, sld As Slide
    Dim nomeA As String, hpA As Long, hpMaxA As Long, lvA As Integer, idBaseA As Long
    Dim nomeB As String, hpB As Long, hpMaxB As Long, lvB As Integer, idBaseB As Long
    Dim turno As String, tipoB As String

    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True)
    
    Call LeggiDatiCompletiBattaglia(xlWorkbook, tipoB, nomeA, lvA, hpA, hpMaxA, idBaseA, nomeB, lvB, hpB, hpMaxB, idBaseB, turno)
    
    xlWorkbook.Close False: Set xlApp = Nothing

    Set sld = ActivePresentation.SlideShowWindow.View.Slide
    If sld Is Nothing Then Exit Sub

    Call AggiornaSpritePokemon(sld, idBaseA, "A")
    Call AggiornaSpritePokemon(sld, idBaseB, "B")
    Call AggiornaUI_PostAzione(sld, nomeA, lvA, hpA, hpMaxA, nomeB, lvB, hpB, hpMaxB)
    
    Call AbilitaPulsantiPerTurno(sld, turno, tipoB, idBaseA, idBaseB)
    
    Exit Sub
GestioneErrori:
    If Not xlApp Is Nothing Then xlApp.Quit
End Sub

Public Sub AggiornaPulsantiMossa(sld As Slide, idBase As Long, Lato As String)
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim pkmnBaseSheet As Object, mosseSheet As Object
    
    ' Array per contenere tutti i dati delle 3 mosse
    Dim nomiMossa(1 To 3) As String
    Dim tipiMossa(1 To 3) As String
    Dim idMossa(1 To 3) As Long
    Dim numDadi(1 To 3) As Integer
    Dim incremento(1 To 3) As Long
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True)
    
    Set pkmnBaseSheet = xlWorkbook.Sheets("Pokemon_Base")
    Set mosseSheet = xlWorkbook.Sheets("Mosse")
    
    ' 1. RECUPERA GLI ID DELLE MOSSE DEL POKEMON
    Dim rigaPkmn As Range, livelloTemp As Integer
    Set rigaPkmn = pkmnBaseSheet.Columns("A").Find(what:=idBase, lookIn:=-4163, lookAt:=1)
    If Not rigaPkmn Is Nothing Then
        idMossa(1) = CLng(val(pkmnBaseSheet.Cells(rigaPkmn.Row, "F").value))
        idMossa(2) = CLng(val(pkmnBaseSheet.Cells(rigaPkmn.Row, "G").value))
        idMossa(3) = CLng(val(pkmnBaseSheet.Cells(rigaPkmn.Row, "H").value))
    End If
    
    ' 2. RECUPERA I DETTAGLI PER OGNI MOSSA ESISTENTE
    Dim i As Integer, istanza As Long
    If Lato = "A" Then istanza = xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value Else istanza = xlWorkbook.Sheets("Battaglia_Corrente").Cells(7, 2).value
    Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, istanza, idBase, nomiMossa(1), livelloTemp)
            
    For i = 1 To 3
        If idMossa(i) > 0 Then
            Call Mod_Utilities.OttieniParametriMossaAlLivello(xlWorkbook, idMossa(i), livelloTemp, nomiMossa(i), tipiMossa(i), numDadi(i), incremento(i))
        End If
    Next i
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
    
    ' 3. AGGIORNA L'INTERFACCIA SULLA SLIDE
    Dim nomePulsante As String, nomeTestoDado As String, nomeTestoIncremento As String
    Dim nomeIconaDado As String, nomeGruppoIcone As String, nomeIconaSpecifica As String, nomeIconaBase As String
    
    ' === RESET PREVENTIVO DI TUTTE LE ICONE DI TIPO ===
    Dim k As Integer, tipoIcona As Variant
    For k = 1 To 3
        For Each tipoIcona In Array("Fuoco", "Acqua", "Erba", "Normale", "Elettro", "Terra", "Oscurità", "Psico")
            On Error Resume Next
            sld.Shapes("Icona_Mossa" & k & "_" & tipoIcona).Visible = msoFalse
            sld.Shapes("Icona_Mossa" & k & "_B_" & tipoIcona).Visible = msoFalse
            On Error GoTo 0
        Next tipoIcona
    Next k
    ' ===================================================

    ' Ciclo principale per popolare e mostrare/nascondere
    For i = 1 To 3
        ' Determina i nomi delle forme
        nomePulsante = "PulsanteMossa" & i
        nomeTestoDado = "TestoDadoMossa" & i
        nomeTestoIncremento = "TestoIncrementoMossa" & i
        nomeIconaDado = "IconaDadoMossa" & i
        nomeGruppoIcone = "Icone_Mossa" & i
        
        If Lato = "B" Then
            nomePulsante = nomePulsante & "_B"
            nomeTestoDado = nomeTestoDado & "_B"
            nomeTestoIncremento = nomeTestoIncremento & "_B"
            nomeIconaDado = nomeIconaDado & "_B"
            nomeGruppoIcone = nomeGruppoIcone & "_B"
        End If
        
        On Error Resume Next
        
        If idMossa(i) > 0 Then
            ' La mossa esiste: mostra i componenti e popola i dati
            sld.Shapes(nomePulsante).Visible = msoTrue
            sld.Shapes(nomeTestoDado).Visible = msoTrue
            sld.Shapes(nomeTestoIncremento).Visible = msoTrue
            sld.Shapes(nomeIconaDado).Visible = msoTrue
            
            ' Popola i testi
            sld.Shapes(nomePulsante).TextFrame.TextRange.text = nomiMossa(i)
            sld.Shapes(nomeTestoDado).TextFrame.TextRange.text = numDadi(i)
            If incremento(i) > 0 Then sld.Shapes(nomeTestoIncremento).TextFrame.TextRange.text = "+" & incremento(i) Else sld.Shapes(nomeTestoIncremento).TextFrame.TextRange.text = "+0"
            
            ' Mostra l'icona di tipo corretta
            If Trim(tipiMossa(i)) <> "" Then
                nomeIconaBase = "Icona_Mossa" & i
                If Lato = "B" Then nomeIconaBase = nomeIconaBase & "_B"
                nomeIconaSpecifica = nomeIconaBase & "_" & tipiMossa(i)
                sld.Shapes(nomeIconaSpecifica).Visible = msoTrue
            End If
            
        Else
            ' La mossa NON esiste: nascondi tutti i componenti
            sld.Shapes(nomePulsante).Visible = msoFalse
            sld.Shapes(nomeTestoDado).Visible = msoFalse
            sld.Shapes(nomeTestoIncremento).Visible = msoFalse
            sld.Shapes(nomeIconaDado).Visible = msoFalse
            ' Nascondiamo anche il gruppo di icone per sicurezza, anche se le icone interne sono già state nascoste dal reset
            sld.Shapes(nomeGruppoIcone).Visible = msoFalse
        End If
        
        On Error GoTo 0
    Next i
End Sub

Public Sub AggiornaSpritePokemon(sld As Slide, ID_Base As Long, Lato As String)
    Dim gruppoSprite As Shape, sprite As Shape
    Dim nomeGruppo As String, nomeSpriteDaMostrare As String
    
    If Lato = "A" Then
        nomeGruppo = "Sprite_A_Back"
    Else
        nomeGruppo = "Sprite_B_Front"
    End If
    
    nomeSpriteDaMostrare = "Sprite_" & Lato & "_" & ID_Base
    
    On Error Resume Next
    Set gruppoSprite = Nothing
    Set gruppoSprite = sld.Shapes(nomeGruppo)
    If Not gruppoSprite Is Nothing Then
        Dim gi As Shape
        For Each gi In gruppoSprite.GroupItems
            gi.Visible = msoFalse
        Next gi
    End If
    On Error GoTo 0
    
    On Error Resume Next
    Dim shpToShow As Shape
    Set shpToShow = sld.Shapes(nomeSpriteDaMostrare)
    If Not shpToShow Is Nothing Then shpToShow.Visible = msoTrue
    On Error GoTo 0
    
    On Error GoTo 0
End Sub

Public Sub AggiornaUI_PostAzione(sld As Slide, nomeA As String, lvA As Integer, hpA As Long, hpMaxA As Long, nomeB As String, lvB As Integer, hpB As Long, hpMaxB As Long)
    Const LARGHEZZA_MAX_HP_A As Single = 240
    Const LARGHEZZA_MAX_HP_B As Single = 240
    On Error Resume Next
    With sld.Shapes
        .Item("Nome_A").TextFrame.TextRange.text = nomeA: .Item("Livello_A").TextFrame.TextRange.text = "LV. " & lvA
        .Item("TestoHP_A").TextFrame.TextRange.text = hpA & "/" & hpMaxA
        If hpMaxA > 0 Then .Item("BarraHP_A").Width = (hpA / hpMaxA) * LARGHEZZA_MAX_HP_A Else .Item("BarraHP_A").Width = 0
        
        .Item("Nome_B").TextFrame.TextRange.text = nomeB: .Item("Livello_B").TextFrame.TextRange.text = "LV. " & lvB
        .Item("TestoHP_B").TextFrame.TextRange.text = hpB & "/" & hpMaxB
        If hpMaxB > 0 Then .Item("BarraHP_B").Width = (hpB / hpMaxB) * LARGHEZZA_MAX_HP_B Else .Item("BarraHP_B").Width = 0
    End With
    On Error GoTo 0
End Sub

' SOSTITUISCI QUESTA FUNCTION in Mod_UI_Manager
Public Function GetSlideIDPostBattaglia() As Long
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim luogoRitorno As String, mappeSheet As Object, cellaTrovata As Range
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True) ' Sola lettura
    
    Set mappeSheet = xlWorkbook.Sheets("Mappe")
    luogoRitorno = xlWorkbook.Sheets("Battaglia_Corrente").Cells(11, 2).value ' Leggi Fonte_Ritorno
    
    ' Controlla che il luogo non sia vuoto
    If Trim(luogoRitorno) = "" Then
        GetSlideIDPostBattaglia = 3 ' Default di sicurezza
        GoTo Cleanup
    End If

    ' Cerca il luogo nel foglio Mappe
    Set cellaTrovata = mappeSheet.Columns("A").Find(what:=luogoRitorno, lookIn:=-4163, lookAt:=1)
    
    If Not cellaTrovata Is Nothing Then
        ' Trovato! Leggi l'ID dalla colonna accanto (B)
        GetSlideIDPostBattaglia = CLng(mappeSheet.Cells(cellaTrovata.Row, "B").value)
    Else
        ' NON Trovato! Avvisa l'utente e usa il default
        MsgBox "ATTENZIONE: Luogo di ritorno '" & luogoRitorno & "' non trovato nel foglio 'Mappe'. Ritorno alla mappa principale.", vbExclamation
        GetSlideIDPostBattaglia = 3 ' Default di sicurezza
    End If

Cleanup:
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
End Function

Public Sub InfoBox_Click_Handler()
    Dim sld As Slide, messaggio As String, xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim evoluzioneInAttesa As Boolean
    
    Set sld = ActivePresentation.SlideShowWindow.View.Slide
    If sld Is Nothing Then Exit Sub
    
    If sld.Shapes("Pulsante_Prosegui").Visible = msoTrue Or sld.Shapes("Pulsante_Turno_Avversario").Visible = msoTrue Then
        Exit Sub
    End If

    messaggio = sld.Shapes("InfoBox").TextFrame.TextRange.text
    sld.Shapes("InfoBox").Visible = msoFalse
    
    ' CONTROLLA SE C'È UN'EVOLUZIONE IN ATTESA
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile)
    If xlWorkbook.Sheets("Battaglia_Corrente").Cells(22, 2).value = 1 Then
        evoluzioneInAttesa = True
    End If
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
    
    ' DECIDI IL PROSSIMO PASSO
    If evoluzioneInAttesa Then
        ' C'è un'evoluzione: vai alla scena dell'evoluzione
        Call Mod_Game_Events.AvviaScenaEvoluzione
    ElseIf InStr(1, messaggio, "non può più combattere") > 0 Then
        ' Il Pokémon del giocatore è KO: apri la schermata di scambio
        Call Mod_UI_Manager.MostraSchermataScambio
    Else
        ' Altrimenti, è un turno normale: aggiorna l'interfaccia per il prossimo turno
        Call Mod_UI_Manager.AggiornaInterfacciaBattaglia_DaTrigger
    End If
End Sub

' SOSTITUISCI QUESTA SUB in Mod_UI_Manager
Public Sub MostraInfoBox(sld As Slide, messaggio As String, fineBattaglia As Boolean, ByVal turnoProssimo As String, ByVal tipoBattaglia As String)
    Dim infoBoxShape As Shape
    
    On Error Resume Next
    Set infoBoxShape = sld.Shapes("InfoBox")
    If infoBoxShape Is Nothing Then Exit Sub ' Esce se l'InfoBox non esiste
    
    ' 1. Resetta tutti i pulsanti di azione
    Call AbilitaPulsantiPerTurno(sld, "NESSUNO", "", 0, 0)
    
    ' 2. Prepara la InfoBox MENTRE È ANCORA NASCOSTA
    infoBoxShape.Visible = msoFalse
    infoBoxShape.TextFrame.TextRange.text = messaggio

    ' "Scuoti" leggermente la forma per costringere PowerPoint a ricalcolarne i confini
    infoBoxShape.IncrementLeft 0.01
    infoBoxShape.IncrementLeft -0.01
    DoEvents ' Concedi a PowerPoint un istante per processare le modifiche
    ' =============================================================

    ' 3. Ora che è stata ridisegnata correttamente, rendila visibile
    infoBoxShape.Visible = msoTrue
    
    ' 4. DECIDI QUALE PULSANTE DI CONTINUAZIONE MOSTRARE
    If fineBattaglia Then
        sld.Shapes("Pulsante_Prosegui").Visible = msoTrue
    Else
        If InStr(1, messaggio, "non può più combattere") > 0 Then
        ElseIf InStr(1, messaggio, "Appare") > 0 Or InStr(1, messaggio, "Inizia la battaglia") > 0 Then
        Else
            If turnoProssimo = "B" And UCase$(tipoBattaglia) <> "PVP" Then
                sld.Shapes("Pulsante_Turno_Avversario").Visible = msoTrue
            Else
            End If
        End If
    End If
    On Error GoTo 0
End Sub

Public Sub MostraSchermataScambio(Optional ByVal GiocatoreID As Variant)
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String
    Dim sld As Slide, teamSheet As Object, battagliaSheet As Object
    Dim idTarget As Integer, lastRow As Long, i As Integer
    Dim boxSquadra As Shape, slotShape As Shape, spriteShape As Shape, buttonShape As Shape
    Dim hp As Long, pkmnID_Base As Long, slotNum As Integer, nomeSprite As String
    Dim gruppoSprite As Shape, gruppoSpriteKO As Shape
    
    Set sld = ActivePresentation.SlideShowWindow.View.Slide
    If sld Is Nothing Then Exit Sub
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True)
    Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente")
    
    If IsMissing(GiocatoreID) Then
        idTarget = CInt(battagliaSheet.Cells(3, 2).value)
    Else
        idTarget = CInt(GiocatoreID)
    End If
    g_ScambioGiocatoreID = idTarget
    
    If idTarget = 1 Then
        Set teamSheet = xlWorkbook.Sheets("Giocatore1_Squadra")
    Else
        Set teamSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    End If
    
    Set boxSquadra = sld.Shapes("Box_Squadra")
    boxSquadra.Left = (sld.Master.Width / 2) - (boxSquadra.Width / 2)
    boxSquadra.Top = (sld.Master.Height / 2) - (boxSquadra.Height / 2)
    boxSquadra.Visible = msoTrue
    
    On Error Resume Next
    Set gruppoSprite = sld.Shapes("Sprite_Small")
    If Not gruppoSprite Is Nothing Then
        For Each spriteShape In gruppoSprite.GroupItems
            spriteShape.Visible = msoFalse
        Next spriteShape
    End If
    
    Set gruppoSpriteKO = sld.Shapes("Sprite_Small_KO")
    If Not gruppoSpriteKO Is Nothing Then
        For Each spriteShape In gruppoSpriteKO.GroupItems
            spriteShape.Visible = msoFalse
        Next spriteShape
    End If
    
    For i = 1 To 6
        sld.Shapes("Pulsante_Slot_" & i).Visible = msoFalse
    Next i
    On Error GoTo 0

    lastRow = teamSheet.Cells(teamSheet.Rows.count, "A").End(-4162).Row
    For i = 2 To lastRow
        slotNum = i - 1
        pkmnID_Base = CLng(teamSheet.Cells(i, "B").value)
        hp = CLng(teamSheet.Cells(i, "F").value)
        
        On Error Resume Next
        Set slotShape = sld.Shapes("Slot_" & slotNum)
        Set buttonShape = sld.Shapes("Pulsante_Slot_" & slotNum)
        
        If hp <= 0 Then
            Set spriteShape = sld.Shapes("Sprite_Small_KO").GroupItems("Sprite_Small_" & pkmnID_Base & "_KO")
        Else
            Set spriteShape = sld.Shapes("Sprite_Small").GroupItems("Sprite_Small_" & pkmnID_Base)
        End If
        
        If Not slotShape Is Nothing Then
            slotShape.Left = boxSquadra.Left + 30 + ((slotNum - 1) * 80)
            slotShape.Top = boxSquadra.Top + 27
            slotShape.Visible = msoTrue
            
            If Not spriteShape Is Nothing Then
                spriteShape.Left = slotShape.Left + (slotShape.Width / 2) - (spriteShape.Width / 2)
                spriteShape.Top = slotShape.Top + (slotShape.Height / 2) - (spriteShape.Height / 2)
                spriteShape.Visible = msoTrue
                
                If hp > 0 And Not buttonShape Is Nothing Then
                    buttonShape.Left = slotShape.Left
                    buttonShape.Top = slotShape.Top
                    buttonShape.Width = slotShape.Width
                    buttonShape.Height = slotShape.Height
                    buttonShape.Visible = msoTrue
                End If
            End If
        End If
        On Error GoTo 0
    Next i
    
    xlWorkbook.Close False: Set xlApp = Nothing
End Sub

Public Sub TornaAllaMappa()
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, slideID As Long
    Dim luogoRitorno As String, mappeSheet As Object, cellaTrovata As Range

    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile, True) ' Apri in sola lettura
    
    ' Leggi il NOME del luogo dalla cella corretta (B11)
    luogoRitorno = xlWorkbook.Sheets("Battaglia_Corrente").Cells(11, 2).value
    
    If Trim(luogoRitorno) <> "" Then
        Set mappeSheet = xlWorkbook.Sheets("Mappe")
        ' Cerca il nome del luogo nel foglio Mappe
        Set cellaTrovata = mappeSheet.Columns("A").Find(what:=luogoRitorno, lookIn:=-4163, lookAt:=1)
        
        If Not cellaTrovata Is Nothing Then
            ' Trovato! Leggi l'ID della slide dalla colonna B
            slideID = CLng(mappeSheet.Cells(cellaTrovata.Row, "B").value)
        Else
            ' Non trovato, avvisa e usa il default
            MsgBox "ATTENZIONE: Luogo di ritorno '" & luogoRitorno & "' non trovato nel foglio 'Mappe'. Ritorno alla mappa principale.", vbExclamation
            slideID = 3
        End If
    Else
        ' La cella Fonte_Ritorno era vuota, usa il default
        slideID = 3
    End If
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
    
    If slideID > 0 Then Application.SlideShowWindows(1).View.GotoSlide slideID
End Sub

Public Sub GestisciVisibilitaIconeMosse(sld As Slide, stato As Boolean)
    ' Questa sub mostra o nasconde TUTTE le icone delle mosse.
    On Error Resume Next
    Dim i As Integer
    Dim tipoIcona As Variant
    Dim nomeIconaBase As String
    
    ' Se lo stato è FALSE, nascondiamo tutto e usciamo
    If Not stato Then
        For i = 1 To 3
            ' Icone Lato A
            nomeIconaBase = "Icona_Mossa" & i & "_"
            For Each tipoIcona In Array("Fuoco", "Acqua", "Erba", "Normale", "Elettro", "Terra", "Oscurità", "Psico")
                sld.Shapes(nomeIconaBase & tipoIcona).Visible = msoFalse
            Next tipoIcona
            
            ' Icone Lato B
            nomeIconaBase = "Icona_Mossa" & i & "_B_"
            For Each tipoIcona In Array("Fuoco", "Acqua", "Erba", "Normale", "Elettro", "Terra", "Oscurità", "Psico")
                sld.Shapes(nomeIconaBase & tipoIcona).Visible = msoFalse
            Next tipoIcona
        Next i
    End If
    ' Se lo stato è TRUE, la logica di visualizzazione è gestita da AggiornaPulsantiMossa,
    ' quindi non dobbiamo fare nulla qui.
    On Error GoTo 0
End Sub
