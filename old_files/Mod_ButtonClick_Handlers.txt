Public Sub Pulsante_Prosegui_Click()
    ' Questa sub ora chiama direttamente la funzione per tornare alla mappa.
    On Error Resume Next
    With ActivePresentation.SlideShowWindow.View.Slide.Shapes
        .Item("InfoBox").Visible = msoFalse
        .Item("Pulsante_Prosegui").Visible = msoFalse
    End With
    On Error GoTo 0
    
    Call Mod_UI_Manager.TornaAllaMappa
End Sub
