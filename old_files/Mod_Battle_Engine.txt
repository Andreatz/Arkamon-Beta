Option Explicit

Public Sub EseguiAzioneDiBattaglia(Azione As String, Parametro As Integer)
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, sld As Slide
    Dim messaggio As String, battagliaFinita As Boolean, giocatoreSconfitto As Boolean
    Dim msgParts As Object: Set msgParts = CreateObject("System.Collections.ArrayList")
    Dim prossimoTurno As String, tipoBattaglia As String, turnoCorrente As String
    Dim nomeA As String, hpA As Long, hpMaxA As Long, lvA As Integer, idBaseA As Long
    Dim nomeB As String, hpB As Long, hpMaxB As Long, lvB As Integer, idBaseB As Long
    Dim latoAttaccante As String, messaggioFineTurno As String
    Dim messaggioFinale As String
   
    Set sld = ActivePresentation.SlideShowWindow.View.Slide: If sld Is Nothing Then Exit Sub
    If sld.Shapes("InfoBox").Visible = msoTrue And (InStr(1, sld.Shapes("InfoBox").TextFrame.TextRange.text, "VINTO") > 0 Or InStr(1, sld.Shapes("InfoBox").TextFrame.TextRange.text, "SCONFITTO") > 0) Then Exit Sub

    Call Mod_UI_Manager.AbilitaPulsantiPerTurno(sld, "NESSUNO", "", 0, 0)

    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile)
    
    ' --- FASE 1: SETUP E LETTURA DATI INIZIALI ---
    turnoCorrente = xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value
    tipoBattaglia = xlWorkbook.Sheets("Battaglia_Corrente").Cells(2, 2).value
    latoAttaccante = turnoCorrente
    
    Dim nomeAtt As String, statoAtt As String, turniStatoAtt As Integer
    Dim idBaseAtt As Long, lvAtt As Integer
    If latoAttaccante = "A" Then
        Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value, idBaseAtt, nomeAtt, lvAtt)
        statoAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(13, 2).value
        turniStatoAtt = CInt(xlWorkbook.Sheets("Battaglia_Corrente").Cells(14, 2).value)
    Else
        Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, xlWorkbook.Sheets("Battaglia_Corrente").Cells(7, 2).value, idBaseAtt, nomeAtt, lvAtt)
        statoAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(15, 2).value
        turniStatoAtt = CInt(xlWorkbook.Sheets("Battaglia_Corrente").Cells(16, 2).value)
    End If

    ' --- FASE 2: GESTIONE DEGLI STATI CHE PREVENGONO L'AZIONE ---
    Dim skipTurno As Boolean: skipTurno = False
    If turniStatoAtt > 0 Then
        Select Case UCase$(statoAtt)
            Case "ADDORMENTATO"
                Randomize
                If Rnd < 0.5 Then
                    msgParts.Add nomeAtt & " si è svegliato!"
                    Call PulisciStato(latoAttaccante, xlWorkbook) ' Si sveglia, può attaccare
                Else
                    msgParts.Add nomeAtt & " sta dormendo."
                    skipTurno = True ' Dorme, salta il turno
                End If
            Case "CONFUSO"
                Randomize
                If Rnd < 0.3 Then
                    msgParts.Add nomeAtt & " è confuso e si è colpito da solo!"
                    Call CalcolaDannoAutoinflitto(latoAttaccante, xlWorkbook)
                    skipTurno = True ' Si colpisce, salta il turno
                End If
        End Select
    End If

    ' Decremento durata stati
    If turniStatoAtt > 0 And UCase$(statoAtt) <> "AVVELENATO" Then
        turniStatoAtt = turniStatoAtt - 1
        If turniStatoAtt <= 0 Then
            If Len(messaggioFineTurno) > 0 Then messaggioFineTurno = messaggioFineTurno & ". "
            messaggioFineTurno = messaggioFineTurno & nomeAtt & " non è più " & LCase(statoAtt) & "!"
            Call PulisciStato(latoAttaccante, xlWorkbook)
        Else
            If latoAttaccante = "A" Then xlWorkbook.Sheets("Battaglia_Corrente").Cells(14, 2).value = turniStatoAtt Else xlWorkbook.Sheets("Battaglia_Corrente").Cells(16, 2).value = turniStatoAtt
        End If
    End If
    If Len(messaggioFineTurno) > 0 Then msgParts.Add messaggioFineTurno
    If latoAttaccante = "B" Then
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "A"
    End If
    
    ' --- FASE 3: ESECUZIONE DELL'AZIONE PRINCIPALE (se il turno non è saltato) ---
    Dim messaggioAzione As String
    If Not skipTurno Then
        Select Case Azione
            Case "ATTACCO_GIOCATORE"
                If turnoCorrente = "A" Then Call CalcolaEApplicaDanno("A", Parametro, xlWorkbook, messaggioAzione)
                If turnoCorrente = "B" And UCase$(tipoBattaglia) = "PVP" Then Call CalcolaEApplicaDanno("B", Parametro, xlWorkbook, messaggioAzione)
            
            Case "ATTACCO_AVVERSARIO"
                If turnoCorrente = "B" And UCase$(tipoBattaglia) <> "PVP" Then
                    Dim idxMossa As Integer
                    Dim idBaseDif As Long, nm As String, lvDif As Integer
                    Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value, idBaseDif, nm, lvDif)
                    idxMossa = Mod_Utilities.ScegliMossaIA(xlWorkbook, idBaseAtt, lvAtt, idBaseDif)
                    Call CalcolaEApplicaDanno("B", idxMossa, xlWorkbook, messaggioAzione)
                End If
        End Select
    End If
    If Len(messaggioAzione) > 0 Then msgParts.Add messaggioAzione

    If skipTurno Then
        If latoAttaccante = "A" Then
            xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "B"
        Else
            xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "A"
        End If
    End If
    
    ' --- FASE 4: GESTIONE DEGLI EFFETTI DI FINE TURNO (Veleno e decremento stati) ---
    ' Leggi di nuovo lo stato perché potrebbe essere cambiato
    If latoAttaccante = "A" Then statoAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(13, 2).value: turniStatoAtt = CInt(xlWorkbook.Sheets("Battaglia_Corrente").Cells(14, 2).value)
    If latoAttaccante = "B" Then statoAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(15, 2).value: turniStatoAtt = CInt(xlWorkbook.Sheets("Battaglia_Corrente").Cells(16, 2).value)

    ' Danno da veleno
    If UCase$(statoAtt) = "AVVELENATO" Then
        Dim dannoVeleno As Long, hpAtt As Long, hpMaxAtt As Long
        hpMaxAtt = Mod_Utilities.CalcolaHPMax(xlWorkbook, idBaseAtt, lvAtt)
        hpAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(IIf(latoAttaccante = "A", 5, 8), 2).value
        dannoVeleno = Round(hpMaxAtt * 0.1): If dannoVeleno < 1 Then dannoVeleno = 1
        hpAtt = hpAtt - dannoVeleno
        messaggioFineTurno = nomeAtt & " subisce " & dannoVeleno & " danni dal veleno!"
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(IIf(latoAttaccante = "A", 5, 8), 2).value = hpAtt
    End If

    ' --- FASE 5: CONTROLLO FINE BATTAGLIA E AGGIORNAMENTO UI ---
    
    ' Leggiamo lo stato DOPO tutte le azioni del turno
    Call Mod_UI_Manager.LeggiDatiCompletiBattaglia(xlWorkbook, tipoBattaglia, nomeA, lvA, hpA, hpMaxA, idBaseA, nomeB, lvB, hpB, hpMaxB, idBaseB, prossimoTurno)
    
    Dim moneteCambiate As Long
    
    If hpA <= 0 Then
        ' --- GIOCATORE SCONFITTO ---
        giocatoreSconfitto = True
        Call SalvaHPGiocatore(xlWorkbook, 0)
        
        If ControllaSquadraRimanente(xlWorkbook) Then
            msgParts.Add nomeA & " non può più combattere! Scegli il prossimo."
            battagliaFinita = False ' La partita non finisce, si deve scambiare
        Else
            msgParts.Add "SEI STATO SCONFITTO!"
            battagliaFinita = True
            moneteCambiate = CalcolaVariazioneMonete(False, xlWorkbook)
            If moneteCambiate <> 0 Then msgParts.Add "Hai perso " & Abs(moneteCambiate) & " monete."
            Call TerminaBattaglia(False, xlWorkbook)
        End If
        
    ElseIf hpB <= 0 Then
        Dim pkmnAttivoSheet As Object, pkmnAttivoRiga As Range
        If latoAttaccante = "A" Then
            If xlWorkbook.Sheets("Battaglia_Corrente").Cells(3, 2).value = 1 Then Set pkmnAttivoSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set pkmnAttivoSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
        Else ' L'attaccante era B, ma l'EXP va comunque al Pokémon attivo di A
             If xlWorkbook.Sheets("Battaglia_Corrente").Cells(3, 2).value = 1 Then Set pkmnAttivoSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set pkmnAttivoSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
        End If
        Set pkmnAttivoRiga = pkmnAttivoSheet.Columns("A").Find(what:=xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value)
        
        ' ASSEGNA 1 PUNTO EXP
        If Not pkmnAttivoRiga Is Nothing Then
            pkmnAttivoSheet.Cells(pkmnAttivoRiga.Row, "G").value = pkmnAttivoSheet.Cells(pkmnAttivoRiga.Row, "G").value + 1
            msgParts.Add nomeA & " ha guadagnato 1 punto EXP!"
            
            ' CONTROLLA IL LEVEL UP
            Dim messaggioLV As String
            messaggioLV = Mod_Utilities.GestisciLevelUp(xlWorkbook, pkmnAttivoSheet, pkmnAttivoRiga.Row)
            If messaggioLV <> "" Then
                msgParts.Add messaggioLV
                Call Mod_UI_Manager.LeggiDatiCompletiBattaglia(xlWorkbook, tipoBattaglia, nomeA, lvA, hpA, hpMaxA, idBaseA, nomeB, lvB, hpB, hpMaxB, idBaseB, prossimoTurno)
            End If
        End If

        ' AVVERSARIO SCONFITTO
        If tipoBattaglia = "Selvatico" Then
            msgParts.Add "HAI VINTO LA BATTAGLIA!"
            battagliaFinita = True
            moneteCambiate = CalcolaVariazioneMonete(True, xlWorkbook)
            If moneteCambiate > 0 Then msgParts.Add "Hai ottenuto " & moneteCambiate & " monete."
            Call TerminaBattaglia(True, xlWorkbook)
        Else ' È un allenatore
            Dim idAllenatoreB As Long, indiceAttualeB As Long, ko_csv As String
            idAllenatoreB = xlWorkbook.Sheets("Battaglia_Corrente").Cells(6, 2).value
            indiceAttualeB = xlWorkbook.Sheets("Battaglia_Corrente").Cells(17, 2).value
            ko_csv = CStr(xlWorkbook.Sheets("Battaglia_Corrente").Cells(18, 2).value) ' Forza la conversione a String
            
            ko_csv = Mod_Utilities.CSVAppendUnique(ko_csv, CStr(indiceAttualeB))
            xlWorkbook.Sheets("Battaglia_Corrente").Cells(18, 2).value = ko_csv
            
            Dim nuovoID_Base As Long, nuovoLV As Integer, nuovoIndice As Long
            If Mod_Utilities.ScegliProssimoPokemonAllenatore(xlWorkbook, idAllenatoreB, tipoBattaglia, ko_csv, nuovoID_Base, nuovoLV, nuovoIndice) Then
                ' Ha un altro Pokémon
                msgParts.Add nomeB & " è stato sconfitto!"
                Dim nuovoNome As String, idBaseTemp As Long, lvTemp As Integer
                Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, nuovoID_Base, idBaseTemp, nuovoNome, lvTemp)
                
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(7, 2).value = nuovoID_Base
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(8, 2).value = Mod_Utilities.CalcolaHPMax(xlWorkbook, nuovoID_Base, nuovoLV)
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(9, 2).value = nuovoLV
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(17, 2).value = nuovoIndice
                
                ' CORREZIONE TURNO: Il turno passa al nuovo Pokémon dell'avversario
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "B"
                battagliaFinita = False
                msgParts.Add xlWorkbook.Sheets("Battaglia_Corrente").Cells(10, 2).value & " manda in campo " & nuovoNome & "!"

                ' Rileggi i dati appena scritti per aggiornare le variabili locali
                Call Mod_UI_Manager.LeggiDatiCompletiBattaglia(xlWorkbook, tipoBattaglia, nomeA, lvA, hpA, hpMaxA, idBaseA, nomeB, lvB, hpB, hpMaxB, idBaseB, prossimoTurno)
                ' Aggiorna fisicamente la slide
                Call Mod_UI_Manager.AggiornaUI_PostAzione(sld, nomeA, lvA, hpA, hpMaxA, nomeB, lvB, hpB, hpMaxB)
                Call Mod_UI_Manager.AggiornaSpritePokemon(sld, idBaseB, "B") ' Aggiorna specificamente lo sprite B
                
                ' Passa il turno al nuovo Pokémon dell'avversario
                xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "B"
                battagliaFinita = False
            Else
                ' Non ha altri Pokémon, vittoria!
                msgParts.Add "HAI VINTO LA BATTAGLIA!"
                battagliaFinita = True
                moneteCambiate = CalcolaVariazioneMonete(True, xlWorkbook)
                If moneteCambiate > 0 Then msgParts.Add "Hai ottenuto " & moneteCambiate & " monete."
                Call TerminaBattaglia(True, xlWorkbook)
            End If
        End If
    Else
        ' NESSUNO SCONFITTO: Passaggio del turno normale
        If latoAttaccante = "A" Then xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "B" Else xlWorkbook.Sheets("Battaglia_Corrente").Cells(12, 2).value = "A"
    End If
    
    ' Rileggi i dati finali per l'UI
    Call Mod_UI_Manager.LeggiDatiCompletiBattaglia(xlWorkbook, tipoBattaglia, nomeA, lvA, hpA, hpMaxA, idBaseA, nomeB, lvB, hpB, hpMaxB, idBaseB, prossimoTurno)
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, True
    
    messaggioFinale = Join(msgParts.ToArray, vbCrLf)
    
    Call Mod_UI_Manager.AggiornaUI_PostAzione(sld, nomeA, lvA, hpA, hpMaxA, nomeB, lvB, hpB, hpMaxB)
    Call Mod_UI_Manager.MostraInfoBox(sld, messaggioFinale, battagliaFinita, prossimoTurno, tipoBattaglia)
End Sub

Private Sub CalcolaEApplicaDanno(latoAttaccante As String, NumeroMossa As Integer, ByVal xlWorkbook As Object, ByRef out_messaggio As String)
    Dim battagliaSheet As Object, pkmnBaseSheet As Object, mosseSheet As Object
    Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente"): Set pkmnBaseSheet = xlWorkbook.Sheets("Pokemon_Base"): Set mosseSheet = xlWorkbook.Sheets("Mosse")
    
    Dim istanzaAtt As Long, istanzaDif As Long, hpDif As Long, hpAtt As Long, hpMaxAtt As Long
    Dim idBaseAtt As Long, nomeAtt As String, livelloAtt As Integer
    Dim idBaseDif As Long, nomeDif As String, livelloDif As Integer, tipoDif As String ' Aggiunto nomeDif, livelloDif
    Dim mossaID As Long, mossaNome As String, mossaTipo As String, mossaEffetto As String, mossaValore As Double
    Dim numD6 As Integer, incremento As Long
    
    Dim msgParts As Object: Set msgParts = CreateObject("System.Collections.ArrayList")

    ' 1. Determina Attaccante e Difensore
    If latoAttaccante = "A" Then
        istanzaAtt = battagliaSheet.Cells(4, 2).value: istanzaDif = battagliaSheet.Cells(7, 2).value
        hpAtt = battagliaSheet.Cells(5, 2).value: hpDif = battagliaSheet.Cells(8, 2).value
    Else
        istanzaAtt = battagliaSheet.Cells(7, 2).value: istanzaDif = battagliaSheet.Cells(4, 2).value
        hpAtt = battagliaSheet.Cells(8, 2).value: hpDif = battagliaSheet.Cells(5, 2).value
    End If
    
    ' 2. Recupera dettagli di ENTRAMBI i Pokémon
    Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, istanzaAtt, idBaseAtt, nomeAtt, livelloAtt)
    Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, istanzaDif, idBaseDif, nomeDif, livelloDif) ' Recupera anche i dettagli del difensore
    hpMaxAtt = Mod_Utilities.CalcolaHPMax(xlWorkbook, idBaseAtt, livelloAtt)
    
    ' CORREZIONE CHIAVE: Usa la funzione helper per ottenere il tipo del difensore dal suo ID_Base
    tipoDif = Mod_Utilities.GetTipoFromIDBase(xlWorkbook, idBaseDif)

    ' 3. Recupera TUTTI i parametri della mossa
    Dim rigaMossaPkmn As Object: Set rigaMossaPkmn = pkmnBaseSheet.Columns("A").Find(what:=idBaseAtt, lookIn:=-4163, lookAt:=1)
    If rigaMossaPkmn Is Nothing Then out_messaggio = nomeAtt & " è confuso!": Exit Sub
    Select Case NumeroMossa: Case 1: mossaID = rigaMossaPkmn.Offset(0, 5).value: Case 2: mossaID = rigaMossaPkmn.Offset(0, 6).value: Case 3: mossaID = rigaMossaPkmn.Offset(0, 7).value: End Select

    If Not Mod_Utilities.OttieniParametriMossaAlLivello(xlWorkbook, mossaID, livelloAtt, mossaNome, mossaTipo, numD6, incremento) Then
        mossaNome = "Scontro": mossaTipo = "Normale": numD6 = 1: incremento = 0: mossaEffetto = ""
    Else
        Dim rigaMossa As Range: Set rigaMossa = mosseSheet.Columns("A").Find(what:=mossaID)
        Dim colEffetto As Long, colValore As Long
        colEffetto = Mod_Utilities.TrovaColonnaHeader(mosseSheet, "Effetto")
        colValore = Mod_Utilities.TrovaColonnaHeader(mosseSheet, "Valore_Effetto")
        
        If colEffetto > 0 And Not IsEmpty(mosseSheet.Cells(rigaMossa.Row, colEffetto).value) Then
            mossaEffetto = UCase$(CStr(mosseSheet.Cells(rigaMossa.Row, colEffetto).value))
        End If
        If colValore > 0 And IsNumeric(mosseSheet.Cells(rigaMossa.Row, colValore).value) Then
            mossaValore = CDbl(mosseSheet.Cells(rigaMossa.Row, colValore).value)
        End If
    End If
    
    msgParts.Add nomeAtt & " usa " & mossaNome
    
    ' 4. GESTISCI GLI EFFETTI SPECIALI (se presenti)
    If mossaEffetto <> "" Then
        Select Case mossaEffetto
            Case "CURA"
                Dim curaEffettuata As Long, hpDopoCura As Long
                curaEffettuata = Round(hpMaxAtt * (mossaValore / 100))
                hpDopoCura = hpAtt + curaEffettuata
                If hpDopoCura > hpMaxAtt Then hpDopoCura = hpMaxAtt
                If latoAttaccante = "A" Then battagliaSheet.Cells(5, 2).value = hpDopoCura Else battagliaSheet.Cells(8, 2).value = hpDopoCura
                msgParts.Add "Recupera " & curaEffettuata & " HP!"
                
                If xlWorkbook.Sheets("Battaglia_Corrente").Cells(IIf(latoAttaccante = "A", 13, 15), 2).value = "AVVELENATO" Then
                    Call PulisciStato(latoAttaccante, xlWorkbook)
                    msgParts.Add (nomeAtt & " è guarito dal veleno!")
                End If
            
            Case "ADDORMENTATO", "CONFUSO", "AVVELENATO"
                Dim latoDifensore As String
                If latoAttaccante = "A" Then latoDifensore = "B" Else latoDifensore = "A"
                
                ' Chiama la nuova funzione e controlla il risultato
                If ApplicaStato(latoDifensore, mossaEffetto, mossaValore, xlWorkbook) Then
                    ' Ha funzionato!
                    msgParts.Add "L'avversario è ora " & LCase(mossaEffetto) & "!"
                Else
                    ' Non ha funzionato (era già affetto o immune)
                    ' Non aggiungiamo nessun messaggio qui, la logica finale lo gestirà
                End If
        End Select
    End If
    
    ' 5. CALCOLA IL DANNO (se la mossa ne ha)
   If numD6 > 0 Or incremento > 0 Then
        Dim moltiplicatoreTipo As Double, sommaDadi As Long, dannoFinale As Long, msgTipo As String
        
        ' Ora moltiplicatoreTipo sarà sempre corretto
        moltiplicatoreTipo = Mod_Utilities.TipoMultiplier(xlWorkbook, mossaTipo, tipoDif)
        
        sommaDadi = Mod_Utilities.LanciaDadi(numD6)
        dannoFinale = Mod_Utilities.RoundIntHalfUp((sommaDadi + incremento) * moltiplicatoreTipo)
        
        If moltiplicatoreTipo > 1 Then msgTipo = " (Superefficace!)"
        If moltiplicatoreTipo < 1 And moltiplicatoreTipo > 0 Then msgTipo = " (Poco efficace...)"
        If moltiplicatoreTipo = 0 Then msgTipo = " (Nessun effetto!)": dannoFinale = 0
        
        Dim hpAggiornati As Long: hpAggiornati = hpDif - dannoFinale
        If hpAggiornati < 0 Then hpAggiornati = 0
        If latoAttaccante = "A" Then battagliaSheet.Cells(8, 2).value = hpAggiornati Else battagliaSheet.Cells(5, 2).value = hpAggiornati
        
        msgParts.Add "Danno: " & CStr(dannoFinale) & msgTipo
    End If

    ' 6. PASSA IL TURNO E FINALIZZA
    If latoAttaccante = "A" Then battagliaSheet.Cells(12, 2).value = "B" Else battagliaSheet.Cells(12, 2).value = "A"

    ' 7. COSTRUISCI IL MESSAGGIO FINALE
    If msgParts.count = 1 Then
        out_messaggio = msgParts(0) & " Ma non succede nulla..."
    Else
        out_messaggio = Join(msgParts.ToArray, ". ")
    End If
End Sub

Private Function ApplicaStato(LatoDaColpire As String, effetto As String, Valore As Double, xlWorkbook As Object) As Boolean
    ' Questa funzione ora restituisce TRUE se lo stato viene applicato, FALSE altrimenti.
    Dim statoAttuale As String, cellaStato As Range, cellaTurni As Range
    
    ' Imposta le celle bersaglio
    If LatoDaColpire = "A" Then
        Set cellaStato = xlWorkbook.Sheets("Battaglia_Corrente").Cells(13, 2)
        Set cellaTurni = xlWorkbook.Sheets("Battaglia_Corrente").Cells(14, 2)
    Else
        Set cellaStato = xlWorkbook.Sheets("Battaglia_Corrente").Cells(15, 2)
        Set cellaTurni = xlWorkbook.Sheets("Battaglia_Corrente").Cells(16, 2)
    End If
    
    statoAttuale = UCase$(cellaStato.value)
    
    ' CONTROLLO 1: Il bersaglio ha già questo stato?
    If statoAttuale = UCase$(effetto) Then
        ApplicaStato = False ' Fallito: ha già questo stato
        Exit Function
    End If
    
    ' CONTROLLO 2: Immunità Sonno/Confusione
    If (statoAttuale = "ADDORMENTATO" And UCase$(effetto) = "CONFUSO") Or _
       (statoAttuale = "CONFUSO" And UCase$(effetto) = "ADDORMENTATO") Then
        ApplicaStato = False ' Fallito: immunità
        Exit Function
    End If
    
    ' Se tutti i controlli passano, applica il nuovo stato
    cellaStato.value = UCase$(effetto)
    cellaTurni.value = Valore
    ApplicaStato = True ' Successo!
End Function

Private Sub CalcolaDannoAutoinflitto(Lato As String, xlWorkbook As Object)
    Dim danno As Long, hpAttuali As Long, cellaHP As Range
    Dim istanzaAtt As Long, idBaseAtt As Long, nomeAtt As String, livelloAtt As Integer
    Dim numD6 As Integer, incremento As Long, mossaNome As String, mossaTipo As String

    If Lato = "A" Then
        istanzaAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value
        Set cellaHP = xlWorkbook.Sheets("Battaglia_Corrente").Cells(5, 2)
    Else
        istanzaAtt = xlWorkbook.Sheets("Battaglia_Corrente").Cells(7, 2).value
        Set cellaHP = xlWorkbook.Sheets("Battaglia_Corrente").Cells(8, 2)
    End If

    Call Mod_Utilities.GetPokemonDetailsFromInstance(xlWorkbook, istanzaAtt, idBaseAtt, nomeAtt, livelloAtt)

    Call Mod_Utilities.OttieniParametriMossaAlLivello(xlWorkbook, 1, livelloAtt, mossaNome, mossaTipo, numD6, incremento)

    danno = Mod_Utilities.LanciaDadi(numD6) + incremento
    If danno < 1 Then danno = 1
    
    hpAttuali = cellaHP.value - danno
    If hpAttuali < 0 Then hpAttuali = 0
    cellaHP.value = hpAttuali
End Sub

Private Function CalcolaVariazioneMonete(vittoria As Boolean, ByVal xlWorkbook As Object) As Long
    Dim moneteDelta As Long: moneteDelta = 0
    Dim battagliaSheet As Object: Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente")
    
    If vittoria Then
        Dim tipoAvversario As String, idAvversario As Long
        tipoAvversario = battagliaSheet.Cells(2, 2).value
        idAvversario = battagliaSheet.Cells(6, 2).value
        
        If tipoAvversario <> "Selvatico" Then
            If idAvversario >= 203 Then moneteDelta = 1000 Else moneteDelta = 200
        End If
    Else ' Sconfitta
        If battagliaSheet.Cells(2, 2).value <> "Selvatico" Then
            moneteDelta = -200
        End If
    End If
    
    CalcolaVariazioneMonete = moneteDelta
End Function

Private Sub PulisciStato(Lato As String, xlWorkbook As Object)
    If Lato = "A" Then
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(13, 2).value = ""
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(14, 2).value = 0
    Else
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(15, 2).value = ""
        xlWorkbook.Sheets("Battaglia_Corrente").Cells(16, 2).value = 0
    End If
End Sub

Private Sub SalvaHPGiocatore(ByVal xlWorkbook As Object, hpDaSalvare As Long)
    Dim teamSheet As Object, idGiocatoreA As Long, idIstanzaA As Long, rigaPkmn As Range
    idGiocatoreA = xlWorkbook.Sheets("Battaglia_Corrente").Cells(3, 2).value
    idIstanzaA = xlWorkbook.Sheets("Battaglia_Corrente").Cells(4, 2).value
    If idGiocatoreA = 1 Then Set teamSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set teamSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    Set rigaPkmn = teamSheet.Columns("A").Find(what:=idIstanzaA)
    If Not rigaPkmn Is Nothing Then
        If hpDaSalvare < 0 Then hpDaSalvare = 0
        teamSheet.Cells(rigaPkmn.Row, "F").value = hpDaSalvare
    End If
End Sub

Private Sub TerminaBattaglia(vittoria As Boolean, ByVal xlWorkbook As Object)
    Dim teamSheet As Object, idGiocatoreA As Long, battagliaSheet As Object, statoGiocatoreSheet As Object
    
    Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente")
    Set statoGiocatoreSheet = xlWorkbook.Sheets("Stato_Giocatore")
    idGiocatoreA = battagliaSheet.Cells(3, 2).value
    
    ' --- LOGICA MONETE ---
    Dim moneteCambiate As Long
    moneteCambiate = CalcolaVariazioneMonete(vittoria, xlWorkbook) ' Chiama la nuova funzione per sapere di quanto cambiare
    
    If moneteCambiate <> 0 Then
        Dim rigaGiocatore As Range, moneteAttuali As Long
        Set rigaGiocatore = statoGiocatoreSheet.Columns("A").Find(what:=idGiocatoreA, lookIn:=-4163, lookAt:=1)
        If Not rigaGiocatore Is Nothing Then
            moneteAttuali = statoGiocatoreSheet.Cells(rigaGiocatore.Row, "E").value
            moneteAttuali = moneteAttuali + moneteCambiate
            If moneteAttuali < 0 Then moneteAttuali = 0
            statoGiocatoreSheet.Cells(rigaGiocatore.Row, "E").value = moneteAttuali
        End If
    End If
    ' --- FINE LOGICA MONETE ---

    Call SalvaHPGiocatore(xlWorkbook, battagliaSheet.Cells(5, 2).value)

    If idGiocatoreA = 1 Then Set teamSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set teamSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    
    Dim lastRow As Long, i As Long
    lastRow = teamSheet.Cells(teamSheet.Rows.count, "A").End(-4162).Row
    For i = 2 To lastRow
        teamSheet.Cells(i, "F").value = teamSheet.Cells(i, "E").value
    Next i

    battagliaSheet.Range("B13:B16").ClearContents
End Sub

' --- SISTEMA DI CATTURA ---

Public Sub EseguiAzioneCattura()
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, sld As Slide
    Dim battagliaSheet As Object, pkmnBaseSheet As Object
    Dim tipoBattaglia As String, tassoCattura As Double, hpMaxB As Long, hpAttualiB As Long
    Dim sogliaSuccesso As Double, roll As Integer, catturaRiuscita As Boolean
    Dim messaggio As String, pkmnID_Base As Long, pkmnNome As String, pkmnLivello As Integer
    Dim activePlayerID As Integer

    Set sld = ActivePresentation.SlideShowWindow.View.Slide: If sld Is Nothing Then Exit Sub

    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    If Dir(percorsoFile) = "" Then Mod_UI_Manager.MostraInfoBox sld, "Errore: Database.xlsx non trovato!", True, "A", "Errore": Exit Sub

    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile)

    On Error GoTo CleanupErr
    Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente")
    Set pkmnBaseSheet = xlWorkbook.Sheets("Pokemon_Base")

    tipoBattaglia = CStr(battagliaSheet.Cells(2, 2).value)
    If tipoBattaglia <> "Selvatico" Then
        messaggio = "Non puoi catturare i Pokémon di un allenatore!"
        Call Mod_UI_Manager.MostraInfoBox(sld, messaggio, False, "A", tipoBattaglia)
        GoTo Cleanup
    End If

    If battagliaSheet.Cells(5, 2).value <= 0 Then
        messaggio = "Il tuo Pokémon non può più combattere! Scegli il prossimo."
        Call Mod_UI_Manager.MostraInfoBox(sld, messaggio, False, "A", tipoBattaglia)
        GoTo Cleanup
    End If

    pkmnID_Base = CLng(battagliaSheet.Cells(7, 2).value)
    hpAttualiB = CLng(battagliaSheet.Cells(8, 2).value)
    pkmnLivello = CInt(battagliaSheet.Cells(9, 2).value)
    pkmnNome = CStr(battagliaSheet.Cells(10, 2).value)
    hpMaxB = Mod_Utilities.CalcolaHPMax(xlWorkbook, pkmnID_Base, pkmnLivello)

    Dim cellID As Object
    Set cellID = pkmnBaseSheet.Columns("A").Find(what:=pkmnID_Base, lookIn:=-4163, lookAt:=1)
    If Not cellID Is Nothing Then
        tassoCattura = val(cellID.Offset(0, 10).value)
        If tassoCattura <= 0 Then tassoCattura = 1
    Else
        tassoCattura = 1
    End If

    roll = Mod_Utilities.LanciaDadi(3)
    sogliaSuccesso = tassoCattura * (3 - (hpAttualiB / hpMaxB))
    catturaRiuscita = (roll <= sogliaSuccesso)

    If catturaRiuscita Then
        messaggio = "Cattura riuscita! " & pkmnNome & " è stato aggiunto. (Tiro=" & roll & " soglia=" & Format(sogliaSuccesso, "0.00") & ")"
        activePlayerID = CInt(battagliaSheet.Cells(3, 2).value)
        Call AggiungiPokemonCatturato(pkmnID_Base, pkmnNome, pkmnLivello, activePlayerID, xlWorkbook)
        CloseDatabaseWorkbook xlApp, xlWorkbook, True
        Call Mod_UI_Manager.MostraInfoBox(sld, messaggio, True, "A", "Selvatico")
    Else
        messaggio = "La Pokéball non ha funzionato! (Tiro=" & roll & " soglia=" & Format(sogliaSuccesso, "0.00") & ")"
        battagliaSheet.Cells(12, 2).value = "B"
        CloseDatabaseWorkbook xlApp, xlWorkbook, True
        Call Mod_UI_Manager.MostraInfoBox(sld, messaggio, False, "B", "Selvatico")
    End If

Cleanup:
    CloseDatabaseWorkbook xlApp, xlWorkbook, False
    Exit Sub

CleanupErr:
    messaggio = "Errore durante la procedura di cattura."
    Call Mod_UI_Manager.MostraInfoBox(sld, messaggio, True, "A", "Errore")
    Resume Cleanup
End Sub

Private Sub AggiungiPokemonCatturato(ID_Base As Long, NomePokemon As String, livello As Integer, GiocatoreID As Integer, ByVal xlWorkbook As Object)
    Dim teamSheet As Object, numPkmnInSquadra As Integer, AlDeposito As Boolean
    If GiocatoreID = 1 Then Set teamSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set teamSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    numPkmnInSquadra = teamSheet.Cells(teamSheet.Rows.count, "A").End(-4162).Row - 1
    If numPkmnInSquadra >= 6 Then AlDeposito = True Else AlDeposito = False
    Call Mod_Game_Events.ScegliPokemonPerGiocatore(ID_Base, NomePokemon, GiocatoreID, livello, xlWorkbook, AlDeposito)
End Sub

Public Sub EseguiScambio(SlotSelezionato As Integer)
    Dim xlApp As Object, xlWorkbook As Object, percorsoFile As String, sld As Slide
    Dim teamSheet As Object, battagliaSheet As Object
    Dim tipoB As String, idTarget As Integer
    Dim rigaScelta As Integer, nuovoIst As Long, nuovoHP As Long

    Set sld = ActivePresentation.SlideShowWindow.View.Slide: If sld Is Nothing Then Exit Sub
    
    rigaScelta = SlotSelezionato + 1
    
    percorsoFile = ActivePresentation.path & "\Database.xlsx"
    Set xlApp = CreateObject("Excel.Application")
    Set xlWorkbook = xlApp.Workbooks.Open(percorsoFile)
    
    Set battagliaSheet = xlWorkbook.Sheets("Battaglia_Corrente")
    tipoB = CStr(battagliaSheet.Cells(2, 2).value)
    idTarget = g_ScambioGiocatoreID
    
    If idTarget = 1 Then Set teamSheet = xlWorkbook.Sheets("Giocatore1_Squadra") Else Set teamSheet = xlWorkbook.Sheets("Giocatore2_Squadra")
    
    nuovoHP = CLng(teamSheet.Cells(rigaScelta, "F").value)
    If nuovoHP <= 0 Then
        CloseDatabaseWorkbook xlApp, xlWorkbook, False
        Exit Sub ' Già gestito dall'interfaccia, ma doppia sicurezza
    End If
    
    nuovoIst = CLng(teamSheet.Cells(rigaScelta, "A").value)
    
    ' Aggiorna il Pokémon in campo
    battagliaSheet.Cells(4, 2).value = nuovoIst
    battagliaSheet.Cells(5, 2).value = nuovoHP
    
    If idTarget = 1 Then ' Assumendo che lo scambio sia sempre per il Giocatore A/lato A
        battagliaSheet.Cells(13, 2).value = "" ' Pulisci Stato_A
        battagliaSheet.Cells(14, 2).value = 0  ' Resetta Turni_Stato_A
    Else
        ' Se in futuro implementerai lo scambio per il Giocatore B
        ' battagliaSheet.Cells(15, 2).Value = "" ' Pulisci Stato_B
        ' battagliaSheet.Cells(16, 2).Value = 0  ' Resetta Turni_Stato_B
    End If

    
    ' Il turno resta al giocatore per permettergli di agire col nuovo Pokémon
    battagliaSheet.Cells(12, 2).value = "A"
    
    CloseDatabaseWorkbook xlApp, xlWorkbook, True
    
    g_ScambioGiocatoreID = 0
    Call Mod_UI_Manager.GestisciVisibilitaInterfacciaScambio(False, sld)
    Call Mod_UI_Manager.AggiornaInterfacciaBattaglia_DaTrigger
End Sub
